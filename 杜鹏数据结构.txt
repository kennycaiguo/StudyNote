var sizeGrid=20;
init();
function(){
	createMap();
}
function createMap(){
	oUl.style.width=cols*(sizeGrid+1)+1+"px";
	for(var i=0;i<map.length;i++){
		var oLi=document.createElement("li");
		oLi.style.width=sizeGrid+"px";
		oLi.style.height=sizeGrid+"px";
		oUl.appendChild(oLi);
		
		if(map[i]==1){
			oLi.className="red";
		}else if(map[i]==2){
			oLi.className="green";
		}else if(map[i]==3){
			oLi.className="blue";
		}
	}
}
